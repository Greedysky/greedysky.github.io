175 QString QJsonDocument::escapeString(const QString &s) const {
176 
177  QString r;
178 
179  Q_FOREACH(QChar ch, s) {
180  switch(ch.toLatin1()) {
181  case '\"': r.append("\\\""); break;
182  case '\\': r.append("\\\\"); break;
183  #if 0
184  case '/': r.append("\\/"); break;
185  #endif
186  case '\b': r.append("\\b"); break;
187  case '\f': r.append("\\f"); break;
188  case '\n': r.append("\\n"); break;
189  case '\r': r.append("\\r"); break;
190  case '\t': r.append("\\t"); break;
191  default:
192  r += ch;
193  break;
194  }
195  }

197  return r;

200 //------------------------------------------------------------------------------
201 // Name: toJson
202 //------------------------------------------------------------------------------
203 QString QJsonDocument::toJson(const QJsonValue &v, JsonFormat format, int indent) const {
204 
205  QString b;
206  QTextStream ss(&b, QIODevice::WriteOnly | QIODevice::Text);
207  bool compact = (format == QJsonDocument::Compact);
208 
209  switch(v.type()) {
210  case QJsonValue::Null:
211  ss << "null";
212  break;
213  case QJsonValue::Bool:
214  ss << (v.toBool() ? "true" : "false");
215  break;
216  case QJsonValue::Double:
217  {
218  double d = v.toDouble();
219  if (qIsFinite(d)) {
220  // +2 to format to ensure the expected precision
221  ss << QByteArray::number(d, 'g', 15 + 2); // ::digits10 is 15
222  } else {
223  ss << "null"; // +INF || -INF || NaN (see RFC4627#section2.4)
224  }
225  }
226  break;
227  case QJsonValue::String:
228  ss << '"' << escapeString(v.toString()) << '"';
229  break;
230  case QJsonValue::Array:
231  {
232  const QJsonArray a = v.toArray();
233  ss << (compact ? "[" : "[\n");
234  if(!a.empty()) {
235  QJsonArray::const_iterator it = a.begin();
236  QJsonArray::const_iterator e = a.end();
237 
238  if (!compact) ss << QByteArray(4*indent, ' ');
239  ss << toJson(*it++, format, indent+1);
240 
241  for(;it != e; ++it) {
242  ss << (compact ? "," : ",\n");
243  if (!compact) ss << QByteArray(4*indent, ' ');
244  ss << toJson(*it, format, indent+1);
245  }
246  }
247  indent--;
248  ss << (compact ? "]" : QString("\n%1]").arg(QString(4*indent, ' ')));
249  }
250  break;
251  case QJsonValue::Object:
252  {
253  const QJsonObject o = v.toObject();
254  ss << (compact ? "{" : "{\n");
255  if(!o.empty()) {
256  QJsonObject::const_iterator it = o.begin();
257  QJsonObject::const_iterator e = o.end();
258 
259  if (!compact) ss << QByteArray(4*indent, ' ');
260  ss << '"' << escapeString(it.key()) << (compact ? "\":" : "\": ") << toJson(it.value(), format, indent+1);
261  ++it;
262  for(;it != e; ++it) {
263  ss << (compact ? "," : ",\n");
264  if (!compact) ss << QByteArray(4*indent, ' ');
265  ss << '"' << escapeString(it.key()) << (compact ? "\":" : "\": ") << toJson(it.value(), format, indent+1);
266  }
267  }
268  indent--;
269  ss << (compact ? "}" : QString("\n%1}").arg(QString(4*indent, ' ')));
270  }
271  break;
272  case QJsonValue::Undefined:
273  Q_ASSERT(0);
274  break;
275  }
276 
277  return b;
278 }
