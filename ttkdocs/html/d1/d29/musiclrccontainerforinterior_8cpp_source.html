507 void MusicLrcContainerForInterior::mouseMoveEvent(QMouseEvent *event)
508 {
509  if(m_mouseLeftPressed && m_lrcAnalysis->isValid())
510  {
511  m_layoutWidget->stop();
512 
513  m_mouseMoved = true;
514  const int offset = QtMouseEventGlobalY(event) - m_mousePressedAt.y();
515  m_mousePressedAt = QtMouseEventGlobalPos(event);
516 
517  m_lrcChangeOffset += offset;
518  if(m_lrcChangeState && offset > 0)
519  {
520  m_lrcChangeState = false;
521  m_lrcChangeOffset = 0;
522  }
523  else if(!m_lrcChangeState && offset < 0)
524  {
525  m_lrcChangeState = true;
526  m_lrcChangeOffset = 0;
527  }
528 
529  if(m_lrcChangeOffset !=0 && m_lrcChangeOffset % LRC_CHANGED_OFFSET_LIMIT == 0)
530  {
531  int index = m_lrcAnalysis->currentIndex();
532  index += m_lrcChangeOffset / LRC_CHANGED_OFFSET_LIMIT;
533  m_lrcChangeOffset = 0;
534 
535  if(index < 0)
536  {
537  index = 0;
538  }
539  else if(index >= m_lrcAnalysis->count())
540  {
541  index = m_lrcAnalysis->count() - m_lrcAnalysis->lineMiddle() + 2;
542  }
543 
544  int value = G_SETTING_PTR->value(MusicSettingManager::LrcSize).toInt();
545  value = (mapLrcSizeProperty(m_lrcChangeDelta) - mapLrcSizeProperty(value)) / 2;
546 
547  m_lrcAnalysis->setCurrentIndex(index);
548  for(int i = 0; i < m_lrcAnalysis->lineMax(); ++i)
549  {
550  m_lrcManagers[i]->setText(m_lrcAnalysis->text(i - value - 1));
