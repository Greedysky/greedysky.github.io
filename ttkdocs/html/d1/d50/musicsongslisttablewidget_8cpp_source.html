795  const int end = currentRow();

797 
798  if(m_playRowIndex == start)
799  {
800  index = end;
801  }
802  else if(m_playRowIndex == end)
803  {
804  index = (start > end) ? (end + 1) : (end - 1);
805  }
806  else
807  {

809  {

811  }

813  {
814  --index;
815  }
816  }
817 
818  Q_EMIT queryMusicIndexSwaped(start, end, index, songs);
819  for(int i = qMin(start, end); i <= qMax(start, end); ++i)

822  {
823  continue; //skip the current play item index, because the play widget just has one item
824  }

826  QHeaderView *headerview = horizontalHeader();
827  item(i, 1)->setText(MusicUtils::Widget::elidedText(font(), songs[i].name(), Qt::ElideRight, headerview->sectionSize(1) - 10));
828  item(i, 5)->setText(songs[i].playTime());
829  }
830 
831  bool state;
832  Q_EMIT isCurrentIndex(state);
833  if(state)
834  {
835  selectRow(index);
