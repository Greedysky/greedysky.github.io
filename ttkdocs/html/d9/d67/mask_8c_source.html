33 __STATIC int Mask_writeFormatInformation(int width, unsigned char *frame, int mask, QRecLevel level)
34 {
35  unsigned int format;
36  unsigned char v;
37  int i;
38  int blacks = 0;
39 
40  format = QRspec_getFormatInfo(mask, level);
41 
42  for(i=0; i<8; i++) {
43  if(format & 1) {
44  blacks += 2;
45  v = 0x85;
46  } else {
47  v = 0x84;
48  }
49  frame[width * 8 + width - 1 - i] = v;
50  if(i < 6) {
51  frame[width * i + 8] = v;
52  } else {
53  frame[width * (i + 1) + 8] = v;
54  }
55  format= format >> 1;
56  }
57  for(i=0; i<7; i++) {
58  if(format & 1) {
59  blacks += 2;
60  v = 0x85;
61  } else {
62  v = 0x84;
63  }
64  frame[width * (width - 7 + i) + 8] = v;
65  if(i == 0) {
66  frame[width * 8 + 7] = v;
67  } else {
68  frame[width * 8 + 6 - i] = v;
69  }
70  format= format >> 1;
71  }
72 
73  return blacks;
74 }

295 unsigned char *Mask_mask(int width, unsigned char *frame, QRecLevel level)
296 {
297  int i;
298  unsigned char *mask, *bestMask;
299  int minDemerit = INT_MAX;
300  int blacks;
301  int bratio;
302  int demerit;
303  int w2 = width * width;
304 
305  mask = (unsigned char *)malloc(w2);
306  if(mask == NULL) return NULL;
307  bestMask = NULL;
308 
309  for(i=0; i<maskNum; i++) {
310 // n1 = n2 = n3 = n4 = 0;
311  demerit = 0;
312  blacks = maskMakers[i](width, frame, mask);
313  blacks += Mask_writeFormatInformation(width, mask, i, level);
314  bratio = (200 * blacks + w2) / w2 / 2; /* (int)(100*blacks/w2+0.5) */
315  demerit = (abs(bratio - 50) / 5) * N4;
316 // n4 = demerit;
317  demerit += Mask_evaluateSymbol(width, mask);
318 // printf("(%d,%d,%d,%d)=%d\n", n1, n2, n3 ,n4, demerit);
319  if(demerit < minDemerit) {
320  minDemerit = demerit;
321  free(bestMask);
322  bestMask = mask;
323  mask = (unsigned char *)malloc(w2);
324  if(mask == NULL) break;
325  }
326  }
327  free(mask);
328  return bestMask;
329 }
