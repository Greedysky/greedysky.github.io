54 {
55 // FTZ_ON;

58  * These have to be kept between calls to this function
59  * hence they are static */
60  static int i = 2, j = 1, k = 0;
61 
62  int index, band, channel;

64 
65 #ifdef BENCHMARK

67 #endif //BENCHMARK

80  for(index = 0; index < samples; index+=nch)
81  {
82  /* For each channel */
83  for(channel = 0; channel < nch; channel++)
84  {
85  pcm[channel] = data[index+channel];
86  /* Preamp gain */
87  pcm[channel] *= preamp[channel];
88 
89  /* add random noise */
90  //pcm[channel] += dither[di];
91 
92  out[channel] = 0.;
93  /* For each band */
94  for(band = 0; band < band_count; band++)
95  {
96  /* Optimization */
97  if(gain[channel][band] > -1.0e-10 && gain[channel][band] < 1.0e-10)
98  continue;
99 
100  /* Store Xi(n) */
101  data_history[channel][band].x[i] = pcm[channel];
102  /* Calculate and store Yi(n) */
103  data_history[channel][band].y[i] =
104  (
105  /* = alpha * [x(n)-x(n-2)] */
106  iir_cf[band].alpha * ( data_history[channel][band].x[i]
107  - data_history[channel][band].x[k])
108  /* + gamma * y(n-1) */
109  + iir_cf[band].gamma * data_history[channel][band].y[j]
110  /* - beta * y(n-2) */
111  - iir_cf[band].beta * data_history[channel][band].y[k]
112  );
113  /*
114  * The multiplication by 2.0 was 'moved' into the coefficients to save
115  * CPU cycles here */
116  /* Apply the gain */
117  out[channel] += data_history[channel][band].y[i]*gain[channel][band]; // * 2.0;
118  } /* For each band */
119 
120  if(eq_options & EQ_TWO_PASSES)
121  {
122  /* Filter the sample again */
123  for(band = 0; band < band_count; band++)
124  {
125  /* Optimization */
126  if(gain[channel][band] > -1.0e-10 && gain[channel][band] < 1.0e-10)
127  continue;
128 
129  /* Store Xi(n) */
130  data_history2[channel][band].x[i] = out[channel];
131  /* Calculate and store Yi(n) */
132  data_history2[channel][band].y[i] =
133  (
134  /* y(n) = alpha * [x(n)-x(n-2)] */
135  iir_cf[band].alpha * (data_history2[channel][band].x[i]
136  - data_history2[channel][band].x[k])
137  /* + gamma * y(n-1) */
138  + iir_cf[band].gamma * data_history2[channel][band].y[j]
139  /* - beta * y(n-2) */
140  - iir_cf[band].beta * data_history2[channel][band].y[k]
141  );
142  /* Apply the gain */
143  out[channel] += data_history2[channel][band].y[i]*gain[channel][band];
144  } /* For each band */
145  }
146 
147  if(eq_options & EQ_CLIP)
148  {
149  /* Volume stuff
150  Scale down original PCM sample and add it to the filters
151  output. This substitutes the multiplication by 0.25
152  Go back to use the floating point multiplication before the
153  conversion to give more dynamic range
154  */
155  out[channel] += pcm[channel]*0.25;
156  data[index+channel] = out[channel] > 1.0 ? 1.0 : (out[channel] < -1.0 ? -1.0 : out[channel]);
157  }
158  else
159  {
160  out[channel] += pcm[channel];
161  data[index+channel] = out[channel];
162  }
163 
164  } /* For each channel */
165 
166  /* Wrap around the indexes */
167  i = (i+1)%3;
168  j = (j+1)%3;
169  k = (k+1)%3;
170 
171  }/* For each pair of samples */
172 
173 #ifdef BENCHMARK
174  timex += get_counter();
175  blength += length;
176  if(count++ == 1024)
177  {
178  printf("FLOATING POINT: %f %d\n",timex/1024.0, blength/1024);
179  blength = 0;
180  timex = 0.;
181  count = 0;
182  }
