159 int QRspec_maximumWords(QRencodeMode mode, int version)
160 {
161  int l;
162  int bits;

164 
165  if(!QRinput_isSplittableMode(mode)) return 0;
166  if(version <= 9) {
167  l = 0;

169  l = 1;
170  } else {
171  l = 2;

235 void QRspec_getEccSpec(int version, QRecLevel level, int spec[5])
236 {
237  int b1, b2;
238  int data, ecc;
239 
240  b1 = eccTable[version][level][0];
241  b2 = eccTable[version][level][1];
242  data = QRspec_getDataLength(version, level);
243  ecc = QRspec_getECCLength(version, level);
244 
245  if(b2 == 0) {
246  spec[0] = b1;
247  spec[1] = data / b1;
248  spec[2] = ecc / b1;
249  spec[3] = spec[4] = 0;
250  } else {
251  spec[0] = b1;
252  spec[1] = data / (b1 + b2);

254  spec[3] = b2;
255  spec[4] = spec[1] + 1;
256  }

311 static void QRspec_putAlignmentPattern(int version, unsigned char *frame, int width)
312 {
313  int d, w, x, y, cx, cy;
314 
315  if(version < 2) return;
316 
317  d = alignmentPattern[version][1] - alignmentPattern[version][0];
318  if(d < 0) {
319  w = 2;
320  } else {
321  w = (width - alignmentPattern[version][0]) / d + 2;
322  }
323 
324  if(w * w - 3 == 1) {
325  x = alignmentPattern[version][0];
326  y = alignmentPattern[version][0];
327  QRspec_putAlignmentMarker(frame, width, x, y);
328  return;
329  }
330 
331  cx = alignmentPattern[version][0];
332  for(x=1; x<w - 1; x++) {
333  QRspec_putAlignmentMarker(frame, width, 6, cx);
334  QRspec_putAlignmentMarker(frame, width, cx, 6);
335  cx += d;
336  }
337 
338  cy = alignmentPattern[version][0];
339  for(y=0; y<w-1; y++) {
340  cx = alignmentPattern[version][0];
341  for(x=0; x<w-1; x++) {
342  QRspec_putAlignmentMarker(frame, width, cx, cy);
343  cx += d;
344  }
345  cy += d;
346  }
347 }

437 static unsigned char *QRspec_createFrame(int version)
438 {
439  unsigned char *frame, *p, *q;
440  int width;
441  int x, y;
442  unsigned int verinfo, v;
443 
444  width = qrspecCapacity[version].width;
445  frame = (unsigned char *)malloc(width * width);
446  if(frame == NULL) return NULL;
447 
448  memset(frame, 0, width * width);
449  /* Finder pattern */
450  putFinderPattern(frame, width, 0, 0);
451  putFinderPattern(frame, width, width - 7, 0);
452  putFinderPattern(frame, width, 0, width - 7);
453  /* Separator */
454  p = frame;
455  q = frame + width * (width - 7);
456  for(y=0; y<7; y++) {
457  p[7] = 0xc0;
458  p[width - 8] = 0xc0;
459  q[7] = 0xc0;
460  p += width;
461  q += width;
462  }
463  memset(frame + width * 7, 0xc0, 8);
464  memset(frame + width * 8 - 8, 0xc0, 8);
465  memset(frame + width * (width - 8), 0xc0, 8);
466  /* Mask format information area */
467  memset(frame + width * 8, 0x84, 9);
468  memset(frame + width * 9 - 8, 0x84, 8);
469  p = frame + 8;
470  for(y=0; y<8; y++) {
471  *p = 0x84;
472  p += width;
473  }
474  p = frame + width * (width - 7) + 8;
475  for(y=0; y<7; y++) {
476  *p = 0x84;
477  p += width;
478  }
479  /* Timing pattern */
480  p = frame + width * 6 + 8;
481  q = frame + width * 8 + 6;
482  for(x=1; x<width-15; x++) {
483  *p = 0x90 | (x & 1);
484  *q = 0x90 | (x & 1);
485  p++;
486  q += width;
487  }
488  /* Alignment pattern */
489  QRspec_putAlignmentPattern(version, frame, width);
490 
491  /* Version information */
492  if(version >= 7) {
493  verinfo = QRspec_getVersionPattern(version);
494 
495  p = frame + width * (width - 11);
496  v = verinfo;
497  for(x=0; x<6; x++) {
498  for(y=0; y<3; y++) {
499  p[width * y + x] = 0x88 | (v & 1);
500  v = v >> 1;
501  }
502  }
503 
504  p = frame + width - 11;
505  v = verinfo;
506  for(y=0; y<6; y++) {
507  for(x=0; x<3; x++) {
508  p[x] = 0x88 | (v & 1);
509  v = v >> 1;
510  }
511  p += width;
512  }
513  }
514  /* and a little bit... */
515  frame[width * (width - 8) + 8] = 0x81;
516 
517  return frame;
518 }
