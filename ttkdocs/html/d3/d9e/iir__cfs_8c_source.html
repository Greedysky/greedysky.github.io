89 #define BETA2(tf0, tf) \
90 (TWOPOWER(GAIN_F1)*TWOPOWER(cos(tf0)) \
91  - 2.0 * TWOPOWER(GAIN_F1) * cos(tf) * cos(tf0) \
92  + TWOPOWER(GAIN_F1) \
93  - TWOPOWER(GAIN_F0) * TWOPOWER(sin(tf)))
94 #define BETA1(tf0, tf) \
95  (2.0 * TWOPOWER(GAIN_F1) * TWOPOWER(cos(tf)) \
96  + TWOPOWER(GAIN_F1) * TWOPOWER(cos(tf0)) \
97  - 2.0 * TWOPOWER(GAIN_F1) * cos(tf) * cos(tf0) \
98  - TWOPOWER(GAIN_F1) + TWOPOWER(GAIN_F0) * TWOPOWER(sin(tf)))
99 #define BETA0(tf0, tf) \
100  (0.25 * TWOPOWER(GAIN_F1) * TWOPOWER(cos(tf0)) \
101  - 0.5 * TWOPOWER(GAIN_F1) * cos(tf) * cos(tf0) \
102  + 0.25 * TWOPOWER(GAIN_F1) \
103  - 0.25 * TWOPOWER(GAIN_F0) * TWOPOWER(sin(tf)))

105 #define GAMMA(beta, tf0) ((0.5 + beta) * cos(tf0))
106 #define ALPHA(beta) ((0.5 - beta)/2.0)

190 /* Find the quadratic root
191  * Always return the smallest root */
192 static int find_root(double a, double b, double c, double *x0) {
193  double k = c-((b*b)/(4.*a));
194  double h = -(b/(2.*a));
195  double x1 = 0.;
196  if(-(k/a) < 0.)
197  return -1;
198  *x0 = h - sqrt(-(k/a));
199  x1 = h + sqrt(-(k/a));
200  if(x1 < *x0)
201  *x0 = x1;
202  return 0;
203 }

205 /* Calculate all the coefficients as specified in the bands[] array */
206 void calc_coeffs()
207 {
208  int i, n;
209  double f1, f2;
210  double x0;
211 
212  n = 0;
213  for(; iir_bands[n].cfs; n++) {
214  double *freqs = (double *)iir_bands[n].cfs;
215  for(i=0; i<iir_bands[n].band_count; i++)
216  {
217 
218  /* Find -3dB frequencies for the center freq */
219  find_f1_and_f2(freqs[i], iir_bands[n].octave, &f1, &f2);
220  /* Find Beta */
221  if( find_root(
222  BETA2(TETA(freqs[i]), TETA(f1)),
223  BETA1(TETA(freqs[i]), TETA(f1)),
224  BETA0(TETA(freqs[i]), TETA(f1)),
225  &x0) == 0)
226  {
227  /* Got a solution, now calculate the rest of the factors */
228  /* Take the smallest root always (find_root returns the smallest one)
229  *
230  * NOTE: The IIR equation is
231  * y[n] = 2 * (alpha*(x[n]-x[n-2]) + gamma*y[n-1] - beta*y[n-2])
232  * Now the 2 factor has been distributed in the coefficients
233  */
234  /* Now store the coefficients */
235  iir_bands[n].coeffs[i].beta = 2.0 * x0;
236  iir_bands[n].coeffs[i].alpha = 2.0 * ALPHA(x0);
237  iir_bands[n].coeffs[i].gamma = 2.0 * GAMMA(x0, TETA(freqs[i]));
238 #ifdef DEBUG
239  printf("Freq[%d]: %f. Beta: %.10e Alpha: %.10e Gamma %.10e\n",
240  i, freqs[i], bands[n].coeffs[i].beta,
241  bands[n].coeffs[i].alpha, bands[n].coeffs[i].gamma);
242 #endif
243  } else {
244  /* Shouldn't happen */
245  iir_bands[n].coeffs[i].beta = 0.;
246  iir_bands[n].coeffs[i].alpha = 0.;
247  iir_bands[n].coeffs[i].gamma = 0.;
248  printf(" **** Where are the roots?\n");
249  }
250  }// for i
251  }//for n
252 }
