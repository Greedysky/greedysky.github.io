102 /* Common code for intializing a Reed-Solomon control block (char or int symbols)
103  * Copyright 2004 Phil Karn, KA9Q
104  * May be used under the terms of the GNU Lesser General Public License (LGPL)
105  */
106 //#undef NULL
107 //#define NULL ((void *)0)
108 
109  int i, j, sr,root,iprim;
110 
111  rs = NULL;
112  /* Check parameter ranges */
113  if(symsize < 0 || symsize > (int)(8*sizeof(data_t))){
114  goto done;
115  }
116 
117  if(fcr < 0 || fcr >= (1<<symsize))
118  goto done;
119  if(prim <= 0 || prim >= (1<<symsize))
120  goto done;
121  if(nroots < 0 || nroots >= (1<<symsize))
122  goto done; /* Can't have more roots than symbol values! */
123  if(pad < 0 || pad >= ((1<<symsize) -1 - nroots))
124  goto done; /* Too much padding */
125 
126  rs = (RS *)calloc(1,sizeof(RS));
127  if(rs == NULL)
128  goto done;
129 
130  rs->mm = symsize;
131  rs->nn = (1<<symsize)-1;
132  rs->pad = pad;
133 
134  rs->alpha_to = (data_t *)malloc(sizeof(data_t)*(rs->nn+1));
135  if(rs->alpha_to == NULL){
136  free(rs);
137  rs = NULL;
138  goto done;
139  }
140  rs->index_of = (data_t *)malloc(sizeof(data_t)*(rs->nn+1));
141  if(rs->index_of == NULL){
142  free(rs->alpha_to);
143  free(rs);
144  rs = NULL;
145  goto done;
146  }
147 
148  /* Generate Galois field lookup tables */
149  rs->index_of[0] = A0; /* log(zero) = -inf */
150  rs->alpha_to[A0] = 0; /* alpha**-inf = 0 */
151  sr = 1;
152  for(i=0;i<rs->nn;i++){
153  rs->index_of[sr] = i;
154  rs->alpha_to[i] = sr;
155  sr <<= 1;
156  if(sr & (1<<symsize))
157  sr ^= gfpoly;
158  sr &= rs->nn;
159  }
160  if(sr != 1){
161  /* field generator polynomial is not primitive! */
162  free(rs->alpha_to);
163  free(rs->index_of);
164  free(rs);
165  rs = NULL;
166  goto done;
167  }
168 
169  /* Form RS code generator polynomial from its roots */
170  rs->genpoly = (data_t *)malloc(sizeof(data_t)*(nroots+1));
171  if(rs->genpoly == NULL){
172  free(rs->alpha_to);
173  free(rs->index_of);
174  free(rs);
175  rs = NULL;
176  goto done;
177  }
178  rs->fcr = fcr;
179  rs->prim = prim;
180  rs->nroots = nroots;
181  rs->gfpoly = gfpoly;
182 
183  /* Find prim-th root of 1, used in decoding */
184  for(iprim=1;(iprim % prim) != 0;iprim += rs->nn)
185  ;
186  rs->iprim = iprim / prim;
187 
188  rs->genpoly[0] = 1;
189  for (i = 0,root=fcr*prim; i < nroots; i++,root += prim) {
190  rs->genpoly[i+1] = 1;
191 
192  /* Multiply rs->genpoly[] by @**(root + x) */
193  for (j = i; j > 0; j--){
194  if (rs->genpoly[j] != 0)
195  rs->genpoly[j] = rs->genpoly[j-1] ^ rs->alpha_to[modnn(rs,rs->index_of[rs->genpoly[j]] + root)];
196  else
197  rs->genpoly[j] = rs->genpoly[j-1];
198  }
199  /* rs->genpoly[0] can never be zero */
200  rs->genpoly[0] = rs->alpha_to[modnn(rs,rs->index_of[rs->genpoly[0]] + root)];
201  }
202  /* convert rs->genpoly[] to index form for quicker encoding */
203  for (i = 0; i <= nroots; i++)
204  rs->genpoly[i] = rs->index_of[rs->genpoly[i]];
205  done:;
206 
207  return rs;
208 }

268 /* The guts of the Reed-Solomon encoder, meant to be #included
269  * into a function body with the following typedefs, macros and variables supplied
270  * according to the code parameters:
271 
272  * data_t - a typedef for the data symbol
273  * data_t data[] - array of NN-NROOTS-PAD and type data_t to be encoded
274  * data_t parity[] - an array of NROOTS and type data_t to be written with parity symbols
275  * NROOTS - the number of roots in the RS code generator polynomial,
276  * which is the same as the number of parity symbols in a block.
277  Integer variable or literal.
278  *
279  * NN - the total number of symbols in a RS block. Integer variable or literal.
280  * PAD - the number of pad symbols in a block. Integer variable or literal.
281  * ALPHA_TO - The address of an array of NN elements to convert Galois field
282  * elements in index (log) form to polynomial form. Read only.
283  * INDEX_OF - The address of an array of NN elements to convert Galois field
284  * elements in polynomial form to index (log) form. Read only.
285  * MODNN - a function to reduce its argument modulo NN. May be inline or a macro.
286  * GENPOLY - an array of NROOTS+1 elements containing the generator polynomial in index form
287 
288  * The memset() and memmove() functions are used. The appropriate header
289  * file declaring these functions (usually <string.h>) must be included by the calling
290  * program.

299 void encode_rs_char(RS *rs, const data_t *data, data_t *parity)
300 {
301  int i, j;
302  data_t feedback;
303 
304  memset(parity,0,NROOTS*sizeof(data_t));
305 
306  for(i=0;i<NN-NROOTS-PAD;i++){
307  feedback = INDEX_OF[data[i] ^ parity[0]];
308  if(feedback != A0){ /* feedback term is non-zero */
309 #ifdef UNNORMALIZED
310  /* This line is unnecessary when GENPOLY[NROOTS] is unity, as it must
311  * always be for the polynomials constructed by init_rs()
312  */
313  feedback = MODNN(NN - GENPOLY[NROOTS] + feedback);
314 #endif
315  for(j=1;j<NROOTS;j++)
316  parity[j] ^= ALPHA_TO[MODNN(feedback + GENPOLY[NROOTS-j])];
317  }
318  /* Shift */
319  memmove(&parity[0],&parity[1],sizeof(data_t)*(NROOTS-1));
320  if(feedback != A0)
321  parity[NROOTS-1] = ALPHA_TO[MODNN(feedback + GENPOLY[0])];
322  else
323  parity[NROOTS-1] = 0;
324  }
325 }
