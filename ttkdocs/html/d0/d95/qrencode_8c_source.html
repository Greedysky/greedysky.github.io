71 static int RSblock_init(RSblock *blocks, int spec[5], unsigned char *data, unsigned char *ecc)
72 {
73  int i;
74  RSblock *block;
75  unsigned char *dp, *ep;
76  RS *rs;
77  int el, dl;
78 
79  dl = QRspec_rsDataCodes1(spec);
80  el = QRspec_rsEccCodes1(spec);
81  rs = init_rs(8, 0x11d, 0, 1, el, 255 - dl - el);
82  if(rs == NULL) return -1;
83 
84  block = blocks;
85  dp = data;
86  ep = ecc;
87  for(i=0; i<QRspec_rsBlockNum1(spec); i++) {
88  RSblock_initBlock(block, dl, dp, el, ep, rs);
89  dp += dl;
90  ep += el;
91  block++;
92  }
93

95 
96  dl = QRspec_rsDataCodes2(spec);
97  el = QRspec_rsEccCodes2(spec);

99  if(rs == NULL) return -1;

101  RSblock_initBlock(block, dl, dp, el, ep, rs);

103  ep += el;
104  block++;
105  }

161 __STATIC unsigned char QRraw_getCode(QRRawCode *raw)
162 {
163  int col, row;
164  unsigned char ret;
165 
166  if(raw->count < raw->dataLength) {
167  row = raw->count % raw->blocks;
168  col = raw->count / raw->blocks;
169  if(col >= raw->rsblock[0].dataLength) {
170  row += raw->b1;
171  }
172  ret = raw->rsblock[row].data[col];
173  } else if(raw->count < raw->dataLength + raw->eccLength) {
174  row = (raw->count - raw->dataLength) % raw->blocks;
175  col = (raw->count - raw->dataLength) / raw->blocks;
176  ret = raw->rsblock[row].ecc[col];
177  } else {
178  return 0;
179  }
180  raw->count++;
181  return ret;
182 }

315 static unsigned char *FrameFiller_next(FrameFiller *filler)
316 {
317  unsigned char *p;
318  int x, y, w;
319 
320  if(filler->bit == -1) {
321  filler->bit = 0;
322  return filler->frame + filler->y * filler->width + filler->x;
323  }
324 
325  x = filler->x;
326  y = filler->y;
327  p = filler->frame;
328  w = filler->width;
329 
330  if(filler->bit == 0) {
331  x--;
332  filler->bit++;
333  } else {
334  x++;
335  y += filler->dir;
336  filler->bit--;
337  }
338 
339  if(filler->dir < 0) {
340  if(y < 0) {
341  y = 0;
342  x -= 2;
343  filler->dir = 1;
344  if(!filler->mqr && x == 6) {
345  x--;
346  y = 9;
347  }
348  }
349  } else {
350  if(y == w) {
351  y = w - 1;
352  x -= 2;
353  filler->dir = -1;
354  if(!filler->mqr && x == 6) {
355  x--;
356  y -= 8;
357  }
358  }
359  }
360  if(x < 0 || y < 0) return NULL;
361 
362  filler->x = x;
363  filler->y = y;
364 
365  if(p[y * w + x] & 0x80) {
366  // This tail recursion could be optimized.
367  return FrameFiller_next(filler);
368  }
369  return &p[y * w + x];
370 }

462 __STATIC QRcode *QRcode_encodeMask(QRinput *input, int mask)
463 {
464  int width, version;
465  QRRawCode *raw;
466  unsigned char *frame, *masked, *p, code, bit;
467  FrameFiller *filler;
468  int i, j;
469  QRcode *qrcode = NULL;
470 
471  if(input->mqr) {
472  errno = EINVAL;
473  return NULL;
474  }
475  if(input->version < 0 || input->version > QRSPEC_VERSION_MAX) {
476  errno = EINVAL;
477  return NULL;
478  }
479  if(input->level > QR_ECLEVEL_H) {
480  errno = EINVAL;
481  return NULL;
482  }
483 
484  raw = QRraw_new(input);
485  if(raw == NULL) return NULL;
486 
487  version = raw->version;
488  width = QRspec_getWidth(version);
489  frame = QRspec_newFrame(version);
490  if(frame == NULL) {
491  QRraw_free(raw);
492  return NULL;
493  }
494  filler = FrameFiller_new(width, frame, 0);
495  if(filler == NULL) {
496  QRraw_free(raw);
497  free(frame);
498  return NULL;
499  }
500 
501  /* inteleaved data and ecc codes */
502  for(i=0; i<raw->dataLength + raw->eccLength; i++) {
503  code = QRraw_getCode(raw);
504  bit = 0x80;
505  for(j=0; j<8; j++) {
506  p = FrameFiller_next(filler);
507  if(p == NULL) goto EXIT;
508  *p = 0x02 | ((bit & code) != 0);
509  bit = bit >> 1;
510  }
511  }
512  QRraw_free(raw);
513  raw = NULL;
514  /* remainder bits */
515  j = QRspec_getRemainder(version);
516  for(i=0; i<j; i++) {
517  p = FrameFiller_next(filler);
518  if(p == NULL) goto EXIT;
519  *p = 0x02;
520  }
521 
522  /* masking */
523  if(mask == -2) { // just for debug purpose
524  masked = (unsigned char *)malloc(width * width);
525  memcpy(masked, frame, width * width);
526  } else if(mask < 0) {
527  masked = Mask_mask(width, frame, input->level);
528  } else {
529  masked = Mask_makeMask(width, frame, mask, input->level);
530  }
531  if(masked == NULL) {
532  goto EXIT;
533  }
534  qrcode = QRcode_new(version, width, masked);
535 
536 EXIT:
537  QRraw_free(raw);
538  free(filler);
539  free(frame);
540  return qrcode;
541 }

543 __STATIC QRcode *QRcode_encodeMaskMQR(QRinput *input, int mask)
544 {
545  int width, version;
546  MQRRawCode *raw;
547  unsigned char *frame, *masked, *p, code, bit;
548  FrameFiller *filler;
549  int i, j;
550  QRcode *qrcode = NULL;
551 
552  if(!input->mqr) {
553  errno = EINVAL;
554  return NULL;
555  }
556  if(input->version <= 0 || input->version > MQRSPEC_VERSION_MAX) {
557  errno = EINVAL;
558  return NULL;
559  }
560  if(input->level > QR_ECLEVEL_Q) {
561  errno = EINVAL;
562  return NULL;
563  }
564 
565  raw = MQRraw_new(input);
566  if(raw == NULL) return NULL;
567 
568  version = raw->version;
569  width = MQRspec_getWidth(version);
570  frame = MQRspec_newFrame(version);
571  if(frame == NULL) {
572  MQRraw_free(raw);
573  return NULL;
574  }
575  filler = FrameFiller_new(width, frame, 1);
576  if(filler == NULL) {
577  MQRraw_free(raw);
578  free(frame);
579  return NULL;
580  }
581 
582  /* inteleaved data and ecc codes */
583  for(i=0; i<raw->dataLength + raw->eccLength; i++) {
584  code = MQRraw_getCode(raw);
585  if(raw->oddbits && i == raw->dataLength - 1) {
586  bit = 1 << (raw->oddbits - 1);
587  for(j=0; j<raw->oddbits; j++) {
588  p = FrameFiller_next(filler);
589  if(p == NULL) goto EXIT;
590  *p = 0x02 | ((bit & code) != 0);
591  bit = bit >> 1;
592  }
593  } else {
594  bit = 0x80;
595  for(j=0; j<8; j++) {
596  p = FrameFiller_next(filler);
597  if(p == NULL) goto EXIT;
598  *p = 0x02 | ((bit & code) != 0);
599  bit = bit >> 1;
600  }
601  }
602  }
603  MQRraw_free(raw);
604  raw = NULL;
605 
606  /* masking */
607  if(mask < 0) {
608  masked = MMask_mask(version, frame, input->level);
609  } else {
610  masked = MMask_makeMask(version, frame, mask, input->level);
611  }
612  if(masked == NULL) {
613  goto EXIT;
614  }
615 
616  qrcode = QRcode_new(version, width, masked);
617 
618 EXIT:
619  MQRraw_free(raw);
620  free(filler);
621  free(frame);
622  return qrcode;
623 }

787  QRcode_List *tail = NULL;

792  if(head == NULL) {
