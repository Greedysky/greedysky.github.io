390  }

394  if(*p_ >= '1' && *p_ <= '9') {
395  while(p_ != end_ && std::isdigit(*p_)) {
396  ++p_;
397  }
398  } else if(*p_ == '0') {
399  ++p_;
400  } else {
401  throwError(QJsonParseError::IllegalNumber);
402  }
403  }
404
