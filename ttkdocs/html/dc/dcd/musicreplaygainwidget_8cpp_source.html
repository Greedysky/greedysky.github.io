304 void MusicReplayGainWidget::analysisOutput()

307  while(m_process->canReadLine())
308  {
309  QByteArray data = m_process->readLine();
310  if(data.contains(GAIN_TRACKDB))
311  {
312  data.replace(GAIN_TRACKDB, QByteArray());
313  data.replace("\r\n", QByteArray());
314  track = QString(data.trimmed());
315  }
316  if(data.contains(GAIN_ALBUMDB))
317  {
318  data.replace(GAIN_ALBUMDB, QByteArray());
319  data.replace("\r\n", QByteArray());
320  album = QString(data.trimmed());
321  }
322  }

324  if(!track.isEmpty() || !album.isEmpty())
325  {
