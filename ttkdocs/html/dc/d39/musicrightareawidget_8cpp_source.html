185 void MusicRightAreaWidget::loadCurrentSongLrc(const QString &name, const QString &path) const
186 {
187  if(checkSettingParameterValue())
188  {
189  m_lrcForInterior->stopDrawLrc();
190  m_lrcForInterior->setCurrentSongName(name);
191 
192  MusicLrcAnalysis::State state;
193  if(FILE_SUFFIX(QFileInfo(path)) == KRC_FILE_PREFIX)
194  {
195  TTK_LOGGER_INFO("Current in krc parser mode");
196  state = m_lrcAnalysis->readFromKrcFile(path);
197  }
198  else
199  {
200  TTK_LOGGER_INFO("Current in lrc parser mode");
201  state = m_lrcAnalysis->readFromLrcFile(path);
202  }
203 
204  m_lrcForInterior->updateCurrentLrc(state);
205  m_lrcForDesktop->stopDrawLrc();
206  m_lrcForDesktop->setCurrentSongName(name);
207 
208  if(state == MusicLrcAnalysis::Failed)
209  {
210  m_lrcForDesktop->updateCurrentLrc(tr("No lrc data file found"), QString(), 0);
211  }
212 
213  if(m_lrcForWallpaper)
214  {
215  m_lrcForWallpaper->stopDrawLrc();
216  m_lrcForWallpaper->setCurrentSongName(name);
217  m_lrcForWallpaper->start(true);
218 
219  if(state == MusicLrcAnalysis::Failed)
220  {
221  m_lrcForWallpaper->updateCurrentLrc(tr("No lrc data file found"));
222  }
223  }
224  }
