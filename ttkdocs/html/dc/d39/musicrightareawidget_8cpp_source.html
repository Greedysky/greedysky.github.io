182  m_lrcForInterior->stopDrawLrc();
183  m_lrcForInterior->setCurrentSongName(name);
184 
185  MusicLrcAnalysis::State state;
186  if(TTK_FILE_SUFFIX(QFileInfo(path)) == KRC_FILE_SUFFIX)
187  {
188  TTK_INFO_STREAM("Current in krc parser mode");
189  state = m_lrcAnalysis->loadFromKrcFile(path);
190  }
191  else
192  {
193  TTK_INFO_STREAM("Current in lrc parser mode");
194  state = m_lrcAnalysis->loadFromLrcFile(path);
195  }
196 
197  m_lrcForInterior->updateCurrentLrc(state);
198  m_lrcForDesktop->stopDrawLrc();
199  m_lrcForDesktop->setCurrentSongName(name);
200 
201  if(state == MusicLrcAnalysis::State::Failed)
202  {
203  m_lrcForDesktop->updateCurrentLrc(tr("No lrc data file found"), {}, 0);
204  }
205 
206  if(m_lrcForWallpaper)
207  {
208  m_lrcForWallpaper->stopDrawLrc();
209  m_lrcForWallpaper->setCurrentSongName(name);
210  m_lrcForWallpaper->start(true);
211 
212  if(state == MusicLrcAnalysis::State::Failed)
213  {
214  m_lrcForWallpaper->updateCurrentLrc(tr("No lrc data file found"));
215  }
